import { NotFoundException } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { FilterDto } from '../dtos/filter.dto';
import Character from '../entities/character.entity';
import Location from '../entities/location.entity';
import { Gender, Status } from '../enum/index.enum';

export default class CharacterService {
  constructor(
    @InjectRepository(Character) private characterModel: Repository<Character>
  ) {}

  async getHello(): Promise<string> {
    return 'Hello !';
  }

  async getCharacters(filterDto: FilterDto) {
    const { gender, status, sort } = filterDto;
    const builder = this.characterModel.createQueryBuilder('character').leftJoinAndSelect('character.location', 'location');
    if (status) {
      builder.andWhere('character.status = :status', { status });
    }

    if (gender) {
      builder.andWhere('character.gender = :gender', { gender });
    }

    if (sort) {
      builder.orderBy('location.name', sort);
    }
    return await builder.getMany();

  }

  async createCharacter(
    firstName: string,
    lastName: string,
    status: Status,
    gender: Gender,
    location: Location,
  ): Promise<Character> {
    const character = this.characterModel.create({
      firstName,
      lastName,
      status,
      gender,
    });
    character.location = location;

    return this.characterModel.save(character);
  }

  async getCharacter(id: number): Promise<Character> {
    const character = await this.characterModel.findOne({
      where: { id },
    });

    if (!character) {
      throw new NotFoundException(`Character with ID "${id}" not found`);
    }

    return character;
  }
}
